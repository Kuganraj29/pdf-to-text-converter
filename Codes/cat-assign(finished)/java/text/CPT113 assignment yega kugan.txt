class Members1 // create class to create stack for the group members 
{ 
 private: // create stack named Group1 
  struct Group1 
{ 
     int members1,members2,members3; 
     Group1 *next;  
}; 
 
        Group1 *stackTop; 
 
    public: // constructor and destructor 
     Members1(){stackTop=nullptr;} 
     ~Members1() 
     { 
      Group1 *temp; 
      while(stackTop != nullptr) 
      { 
       temp = stackTop; // temp is point to the stackTop currentnode 
       stackTop=stackTop->next; // advance stackTop to the next 
       delete temp; // deallocate memory occupied by temp 
   } 
  } 
    
 // function that used within the class 
 bool isEmpty();

};
